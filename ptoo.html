<script>
    // Словарь строительного сленга (симуляция работы нейросети)
    const slangDictionary = {
        "залили": "выполнено бетонирование",
        "заливали": "устройство монолитных конструкций",
        "кубов": "м³",
        "куба": "м³",
        "арматуру связали": "выполнено устройство арматурного каркаса",
        "связали": "вязка арматуры",
        "тонн": "т.",
        "кинули кабель": "произведена прокладка кабельной линии",
        "повесили лотки": "выполнен монтаж металлических кабельных лотков",
        "метров": "п.м.",
        "выкопали": "разработка грунта механизированным способом",
        "закопали": "обратная засыпка грунта",
        "плиту": "фундаментной плиты / плиты перекрытия"
    };

    function processText() {
        let rawText = document.getElementById('rawInput').value.toLowerCase();
        
        if (!rawText.trim()) {
            alert("Пожалуйста, введите текст от прораба.");
            return;
        }

        document.getElementById('loader').style.display = 'block';
        document.getElementById('resultBox').style.display = 'none';

        // Имитация задержки обработки данных
        setTimeout(() => {
            let processedText = rawText;
            let missingData = [];

            // 1. Замена сленга
            for (let [slang, official] of Object.entries(slangDictionary)) {
                let regex = new RegExp("\\b" + slang + "\\b", "gi");
                processedText = processedText.replace(regex, `<span class="highlight">${official}</span>`);
            }

            // 2. Поиск объемов
            let hasNumbers = /\d+/.test(rawText);
            if (!hasNumbers) {
                missingData.push("Точные объемы выполненных работ (м³, т, п.м.)");
            }

            // 3. Поиск осей и отметок
            if (!rawText.includes("осях") && !rawText.includes("оси") && !rawText.includes("отметке")) {
                missingData.push("Привязка к строительным осям и высотным отметкам");
            }

            // Формирование HTML результата
            let finalHtml = `
                <div style="margin-bottom: 10px;"><strong>Формулировка:</strong></div>
                <div>${processedText}</div>
            `;

            // Добавление предупреждений, если не хватает данных
            if (missingData.length > 0) {
                finalHtml += `
                    <div class="warning">
                        <strong>⚠️ Внимание ПТО!</strong> Для корректной записи в ОЖР необходимо уточнить у прораба:
                        <ul style="margin-top: 5px; margin-bottom: 0; color: #333;">
                            ${missingData.map(item => `<li>${item}</li>`).join('')}
                        </ul>
                    </div>
                `;
            }

            document.getElementById('loader').style.display = 'none';
            document.getElementById('resultContent').innerHTML = finalHtml;
            document.getElementById('resultBox').style.display = 'block';

        }, 1000);
    }
</script>
